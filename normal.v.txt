module Normal_Mode(
    input wire clk,
    input wire reset,
    input wire [31:0] TIMER_CNT,
    input wire [31:0] TIMER_TOP,
    output reg interrupt
);

    always @(posedge clk or posedge reset) begin
        if (reset) begin
            interrupt <= 0;
        end else if (TIMER_CNT == TIMER_TOP) begin
            interrupt <= 1;
        end else begin
            interrupt <= 0;
        end
    end

endmodule
