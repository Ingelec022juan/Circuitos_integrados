module Fast_PWM(
    input wire clk,
    input wire reset,
    input wire [31:0] TIMER_CNT,
    input wire [31:0] TIMER_TOP,
    input wire [31:0] PWM_CNTA,
    input wire [31:0] PWM_CNTB,
    output reg PWM_OUTA,
    output reg PWM_OUTB
);

    always @(posedge clk or posedge reset) begin
        if (reset) begin
            PWM_OUTA <= 0;
            PWM_OUTB <= 0;
        end else begin
            PWM_OUTA <= (TIMER_CNT < PWM_CNTA);
            PWM_OUTB <= (TIMER_CNT < PWM_CNTB);
        end
    end

endmodule
